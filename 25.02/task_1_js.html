📘 JavaScript с нуля

Лекция 1 — alert · prompt · confirm  |  40 интерактивных задач

🧠 Теория

Функция Описание Возвращает
alert(msg) Показывает сообщение undefined
prompt(msg) Запрашивает ввод текста строку или null
confirm(msg) Вопрос Да / Нет true или false

⚠️ prompt() всегда возвращает строку. Для арифметики оборачивай в Number().

🟢 Блок 1 — alert: вывод сообщений (1–10)
1
Привет, мир!
Выведи в alert фразу: Привет, мир! 
alert("Привет, мир!");
2
Вывести имя
Выведи своё имя через alert.
alert("Иван");
3
Переменная и alert
Объяви city = "Москва" и выведи через alert. 
let city = "Москва";
alert(city);
4
Полное имя
Объяви firstName и lastName. Выведи полное имя. 
let firstName = "Анна";
let lastName  = "Иванова";
alert(firstName + " " + lastName);
5
Год в сообщении
Объяви year = 2026 и выведи: Сейчас 2026 год. 
let year = 2026;
alert("Сейчас " + year + " год");
6
Шаблонная строка
Объяви country = "Россия". Используй шаблонную строку. 
let country = "Россия";
alert(`Я живу в ${country}`);
7
Сложение в alert
Выведи результат сложения 5 + 3. 
alert(5 + 3);
8
Арифметика
Объяви a = 10, b = 4. Выведи сумму, разность, произведение, частное. 
let a = 10, b = 4;
alert(a + b); // 14
alert(a - b); // 6
alert(a * b); // 40
alert(a / b); // 2.5
9
Остаток от деления
Выведи остаток от деления 17 на 5. 
alert(17 % 5); // 2
10
Итоговая стоимость
Объяви price = 199, count = 3. Выведи: Итого: 597 руб. 
let price = 199, count = 3;
alert("Итого: " + price * count + " руб.");
🟡 Блок 2 — prompt: ввод данных (11–20)
11
Приветствие
Спроси имя пользователя и поздоровайся с ним. 
let name = prompt("Как тебя зовут?");
alert("Привет, " + name + "!");
12
Любимый цвет
Спроси любимый цвет и выведи его в сообщении. 
let color = prompt("Любимый цвет?");
alert("Твой любимый цвет — " + color);
13
Сумма двух чисел
Спроси два числа и выведи их сумму. ⚠️ Используй Number()! 
let a = Number(prompt("Первое число:"));
let b = Number(prompt("Второе число:"));
alert("Сумма: " + (a + b));
14
Имя и фамилия
Спроси имя и фамилию отдельными prompt. Выведи полное имя. 
let first = prompt("Имя:");
let last  = prompt("Фамилия:");
alert("Полное имя: " + first + " " + last);
15
Возраст через 10 лет
Спроси возраст и выведи, сколько будет через 10 лет. 
let age = Number(prompt("Сколько тебе лет?"));
alert("Через 10 лет: " + (age + 10) + " лет");
16
°C → °F
Спроси температуру в Цельсиях и переведи в Фаренгейты. F = C × 1.8 + 32 
let c = Number(prompt("Температура °C:"));
let f = c * 1.8 + 32;
alert(c + "°C = " + f + "°F");
17
Площадь и периметр
Спроси ширину и высоту прямоугольника. Выведи площадь и периметр. 
let w = Number(prompt("Ширина:"));
let h = Number(prompt("Высота:"));
alert("Площадь: " + w * h);
alert("Периметр: " + 2 * (w + h));
18
Минуты → Ч:М
Спроси минуты и переведи в формат Х ч. Х мин. 
let mins = Number(prompt("Минуты:"));
let h = Math.floor(mins / 60);
let m = mins % 60;
alert(mins + " мин = " + h + " ч. " + m + " мин.");
19
Проверка на null
Спроси имя. Если нажали «Отмена» — выведи: Имя не введено. 
let name = prompt("Твоё имя:");
if (name === null) {
  alert("Имя не введено");
} else {
  alert("Привет, " + name + "!");
}
20
Квадрат числа
Спроси число и выведи его квадрат.
let num = Number(prompt("Введи число:"));
alert("Квадрат: " + num * num);
🟠 Блок 3 — confirm: подтверждение (21–28)
21
Ты программист?
Спроси через confirm. Разный ответ в зависимости от выбора. 
let res = confirm("Ты программист?");
if (res) {
  alert("Добро пожаловать в мир кода!");
} else {
  alert("Всё ещё впереди!");
}
22
Выход из системы
Спроси подтверждение выхода. Выведи, что произошло. 
let exit = confirm("Ты уверен, что хочешь выйти?");
if (exit) {
  alert("До свидания!");
} else {
  alert("Остаёмся!");
}
23
Проверка возраста
Через confirm спроси: больше 18? Выведи статус доступа. 
let adult = confirm("Тебе больше 18 лет?");
if (adult) {
  alert("Доступ разрешён");
} else {
  alert("Доступ запрещён");
}
24
Подтверждение имени
Спроси имя через prompt, затем подтверди его через confirm. 
let name = prompt("Как тебя зовут?");
let sure = confirm("Ты точно " + name + "?");
if (sure) {
  alert("Приятно познакомиться, " + name + "!");
} else {
  alert("Хм, странно...");
}
25
Секрет
Спроси: хочешь узнать секрет? Если да — раскрой его. 
let want = confirm("Хочешь узнать секрет?");
if (want) {
  alert("JavaScript — это весело!");
} else {
  alert("Ладно, как хочешь.");
}
26
Показать квадрат?
Спроси число, затем через confirm — показывать ли квадрат. 
let num = Number(prompt("Число:"));
let show = confirm("Показать квадрат " + num + "?");
if (show) {
  alert(num + "² = " + num * num);
}
27
Сменить имя
Спроси имя, предложи его сменить через confirm. 
let name = prompt("Как тебя зовут?");
let ch = confirm("Хочешь другое имя?");
if (ch) {
  name = prompt("Новое имя:");
}
alert("Привет, " + name + "!");
28
Счётчик энтузиазма
3 вопроса через confirm. Подсчитай сколько раз ответил «Да». 
let score = 0;
if (confirm("Любишь JavaScript?"))         score++;
if (confirm("Любишь программировать?"))    score++;
if (confirm("Учишься каждый день?"))       score++;
alert("Энтузиазм: " + score + " из 3");
🔴 Блок 4 — Комбинированные задачи (29–40)
29
Мини-анкета
Спроси имя, возраст и город. Выведи анкету одним alert. 
let name = prompt("Имя:");
let age  = prompt("Возраст:");
let city = prompt("Город:");
alert("Имя: "    + name +
      "\nВозраст: " + age +
      "\nГород: "   + city);
30
Скидка 10%
Спроси цену и количество. Выведи итог и сумму со скидкой 10%. 
let price = Number(prompt("Цена:"));
let count = Number(prompt("Количество:"));
let total = price * count;
alert("Итого: " + total + " руб." +
      "\nСо скидкой: " + total * 0.9 + " руб.");
31
Среднее трёх чисел
Спроси три числа и выведи их среднее арифметическое. 
let a = Number(prompt("Число 1:"));
let b = Number(prompt("Число 2:"));
let c = Number(prompt("Число 3:"));
alert("Среднее: " + (a + b + c) / 3);
32
Расстояние
Спроси скорость (км/ч) и время (ч). Выведи пройденное расстояние. 
let speed = Number(prompt("Скорость (км/ч):"));
let time  = Number(prompt("Время (ч):"));
alert("Расстояние: " + speed * time + " км");
33
Площадь круга
Спроси радиус. Выведи площадь и длину окружности. 
let r  = Number(prompt("Радиус:"));
let pi = 3.14159;
alert("Площадь: " + (pi * r * r).toFixed(2));
alert("Длина:   " + (2 * pi * r).toFixed(2));
34
Секунды → Ч:М:С
Спроси секунды и переведи в формат Ч:М:С. 
let total = Number(prompt("Секунды:"));
let h = Math.floor(total / 3600);
let m = Math.floor((total % 3600) / 60);
let s = total % 60;
alert(h + ":" + m + ":" + s);
35
Новые сообщения
Спроси имя и количество сообщений. Разный вывод для 0 и > 0. 
let name = prompt("Имя:");
let msgs = Number(prompt("Сообщений:"));
if (msgs > 0) {
  alert(name + ", у тебя " + msgs + " сообщений!");
} else {
  alert("Новых сообщений нет.");
}

Спроси пароль. Если "1234" — вход разрешён, иначе — ошибка. 
let pass = prompt("Введи пароль:");
if (pass === "1234") {
  alert("Добро пожаловать!");
} else {
  alert("Неверный пароль.");
}
37
Чётное / нечётное
Спроси число и определи: чётное оно или нечётное. 
let num = Number(prompt("Число:"));
if (num % 2 === 0) {
  alert(num + " — чётное");
} else {
  alert(num + " — нечётное");
}
38
Возрастная группа
Спроси возраст: до 18 — ребёнок, 18–60 — взрослый, старше 60 — пожилой. 
let age = Number(prompt("Возраст:"));
if (age < 18) {
  alert("Ребёнок / подросток");
} else if (age <= 60) {
  alert("Взрослый");
} else {
  alert("Пожилой");
}
39
Время года
Спроси номер месяца (1–12) и определи время года. 
let m = Number(prompt("Месяц (1-12):"));
if      (m >= 3 && m <= 5)  alert("Весна");
else if (m >= 6 && m <= 8)  alert("Лето");
else if (m >= 9 && m <= 11) alert("Осень");
else                         alert("Зима");
40
🏆 Финальная анкета
Собери всё: имя, возраст, статус студента, язык. Выведи красивую анкету. 
let name    = prompt("Имя:");
let age     = Number(prompt("Возраст:"));
let student = confirm("Ты студент?");
let lang    = prompt("Любимый язык программирования:");
let status  = student ? "Студент" : "Не студент";
alert(
  "=== АНКЕТА ===\n" +
  "Имя: "     + name    + "\n" +
  "Возраст: " + age     + "\n" +
  "Статус: "  + status  + "\n" +
  "Язык: "    + lang
);
